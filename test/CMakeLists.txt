function(addtest name)
  add_test(NAME "producttest.${name}"
    COMMAND test/run ${name}
    WORKING_DIRECTORY ${CMAKE_SOURCE_DIR})
  set_tests_properties("producttest.${name}" PROPERTIES ENVIRONMENT CCACHE=${CMAKE_BINARY_DIR}/ccache)
endfunction()
